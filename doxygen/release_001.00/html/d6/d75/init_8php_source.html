251 function error_handler($err_no, $err_str, $err_file, $err_line) {
252  // if error_reporting is set to 0, exit. This is also the case when using @
253  if(ini_get('error_reporting') == '0')
254  return;
255  // handle error
256  switch($err_no) {
257  case E_WARNING:
258  $msg = '[ErrNo=' . $err_no . ' (WARNING), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
259  log_error($msg, (!defined('INSTALLING'))); // e.g. warnings are hidden while installing
260  if(!defined('INSTALLING'))
261  exit;
262  break;
263  case E_USER_ERROR:
264  $msg = '[ErrNo=' . $err_no . ' (USER_ERROR), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
265  log_error($msg);
266  exit;
267  break;
268  case E_USER_WARNING:
269  $msg = $err_str;
270  set_last_message(false, $msg);
271  header('Location: ' . CONST_PROJECT_FULL_URL);
272  exit;
273  break;
274  case E_NOTICE:
275  case E_USER_NOTICE:
276  case 2048: // E_STRICT in PHP5
277  // ignore
278  break;
279  default:
280  // unknown error. Log in file (only) and continue execution
281  $msg = '[ErrNo=' . $err_no . ' (UNKNOWN_ERROR), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
282  log_error($msg, false);
283  break;
284  }
285 }
