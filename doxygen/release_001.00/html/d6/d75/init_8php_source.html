210 function error_handler($err_no, $err_str, $err_file, $err_line) {
211  // if error_reporting is set to 0, exit. This is also the case when using @
212  if(ini_get('error_reporting') == '0')
213  return;
214  // handle error
215  switch($err_no) {
216  case E_WARNING:
217  $msg = '[ErrNo=' . $err_no . ' (WARNING), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
218  log_error($msg, (!defined('INSTALLING'))); // e.g. warnings are hidden while installing
219  if(!defined('INSTALLING'))
220  exit;
221  break;
222  case E_USER_ERROR:
223  $msg = '[ErrNo=' . $err_no . ' (USER_ERROR), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
224  log_error($msg);
225  exit;
226  break;
227  case E_USER_WARNING:
228  $msg = $err_str;
229  set_last_message(false, $msg);
230  header('Location: ' . CONST_PROJECT_FULL_URL);
231  exit;
232  break;
233  case E_NOTICE:
234  case E_USER_NOTICE:
235  case 2048: // E_STRICT in PHP5
236  // ignore
237  break;
238  default:
239  // unknown error. Log in file (only) and continue execution
240  $msg = '[ErrNo=' . $err_no . ' (UNKNOWN_ERROR), File=' . $err_file . ', Line=' . $err_line . '] ' . $err_str;
241  log_error($msg, false);
242  break;
243  }
244 }
